## Bouncing Balls

import P53 from "../p5/P53"

<P53 />

```js
// you can write JSX in your Markdown!
import React from "react"
import Sketch from "react-p5"

class App extends React.Component {
  x = 320
  y = 180
  xspeed = 5
  yspeed = 2

  r = 25

  render() {
    return (
      <div className="App">
        <Sketch
          setup={(p5, parentRef) => {
            p5.createCanvas(640, 360).parent(parentRef)
          }}
          draw={p5 => {
            p5.background(0)
            p5.ellipse(this.x, this.y, this.r * 2, this.r * 2)

            this.x += this.xspeed
            this.y += this.yspeed

            if (this.x > p5.width - this.r || this.x < this.r) {
              this.xspeed = -this.xspeed
            }
            if (this.y > p5.height - this.r || this.y < this.r) {
              this.yspeed = -this.yspeed
            }
          }}
        />
      </div>
    )
  }
}

export default App
```
